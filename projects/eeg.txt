from machine import Pin
from time import sleep

led = Pin(2, Pin.OUT)
btn = Pin(4, Pin.IN, Pin.PULL_UP)

while True:
    if not btn.value():
        led.value(not led.value())
        sleep(0.3)  # debounce delay
